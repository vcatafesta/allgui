/////////////////////////////////////////////////////////////////////
// SPLITBOX
/////////////////////////////////////////////////////////////////////

#xcommand DEFINE SPLITBOX ;
      [ OBJ <obj> ] ;
      [ <dummy1: OF, PARENT> <parent> ] ;
      [ <bottom: BOTTOM> ]   ;
      [ <horizontal: HORIZONTAL> ] ;
      [ <vertical: VERTICAL> ] ;
      [ <rtl: RTL> ] ;
      [ <noattached: NOATTACHED> ] ;
      [ SUBCLASS <subclass> ] ;
   => ;
      [ <obj> := ] _OOHG_SelectSubClass( TSplitBox(), [ <subclass>() ] ): ;
            Define( <(parent)>, <.bottom.>, <.vertical.> .OR. <.horizontal.>, ;
            <.rtl.>, <.noattached.> )

#xcommand END SPLITBOX ;
   => ;
      _EndSplitBox()

#xcommand FORCEBREAK ;
      [ <dummy1: OF, PARENT> <parent> ] ;
   => ;
      _ForceBreak( <(parent)> )
